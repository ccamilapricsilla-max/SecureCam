import React, { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { Play, Activity, Clock, Upload } from 'lucide-react';
import { Button } from '@/components/ui/button';
import { cn } from '@/lib/utils';
import { useCameraContext } from '@/context/CameraContext';
import VideoUploadModal from './VideoUploadModal';

const CameraCard = ({ camera }) => {
  const navigate = useNavigate();
  const { setSelectedCameraId, videoStorage } = useCameraContext();
  const [showUploadModal, setShowUploadModal] = useState(false);
  const isInactive = camera.status === 'INACTIVA';

  // Get data from storage via context helper or direct
  const storedVideo = videoStorage.getVideoByCamera(camera.id);
  const thumbnail = storedVideo?.thumbnail || camera.preview;
  const isStoredVideo = !!storedVideo;

  const handleWatchLive = () => {
    setSelectedCameraId(camera.id);
    navigate('/live');
  };

  const handleUploadClick = (e) => {
    e.stopPropagation();
    setShowUploadModal(true);
  };

  const formatDate = (isoString) => {
    if (!isoString) return 'N/A';
    const date = new Date(isoString);
    return date.toLocaleString('es-ES', { 
      hour: '2-digit', 
      minute: '2-digit',
      day: 'numeric',
      month: 'short'
    });
  };

  return (
    <>
      <VideoUploadModal 
        isOpen={showUploadModal}
        onClose={() => setShowUploadModal(false)}
        cameraId={camera.id}
        onUploadSuccess={() => {}} // Context updates automatically
      />
      
      <div className="group relative overflow-hidden rounded-2xl border border-white/10 bg-gray-900/40 backdrop-blur-sm transition-all duration-300 hover:border-purple-500/30 hover:shadow-[0_0_30px_-5px_rgba(168,85,247,0.15)] flex flex-col h-full">
        {/* Preview Image / Video Thumbnail */}
        <div className="aspect-video w-full overflow-hidden bg-black relative cursor-pointer" onClick={handleWatchLive}>
          <img 
            src={thumbnail} 
            alt={camera.name} 
            className={cn(
              "h-full w-full object-cover transition-transform duration-500 group-hover:scale-110",
              isInactive && "grayscale opacity-50"
            )}
          />
          
          {/* Overlay Gradient */}
          <div className="absolute inset-0 bg-gradient-to-t from-gray-950 via-gray-950/20 to-transparent opacity-80" />

          {/* Status Badge */}
          <div className="absolute top-3 right-3 flex flex-col gap-1 items-end">
            <span className={cn(
              "inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-semibold backdrop-blur-md border shadow-sm",
              isInactive 
                ? "bg-gray-500/20 text-gray-400 border-gray-500/30" 
                : "bg-green-500/20 text-green-400 border-green-500/30 animate-pulse"
            )}>
              <span className={cn("mr-1.5 h-1.5 w-1.5 rounded-full", isInactive ? "bg-gray-400" : "bg-green-400")} />
              {camera.status}
            </span>
            
            {isStoredVideo && (
              <span className="inline-flex items-center rounded-full px-2 py-0.5 text-[10px] font-medium bg-purple-500/20 text-purple-300 border border-purple-500/30 backdrop-blur-md">
                VIDEO CARGADO
              </span>
            )}
          </div>

          {/* Play Button Overlay (Hover) */}
          <div className="absolute inset-0 flex items-center justify-center opacity-0 group-hover:opacity-100 transition-opacity duration-300">
            <div 
              className="h-12 w-12 flex items-center justify-center rounded-full bg-purple-600 hover:bg-purple-700 text-white shadow-lg border border-purple-400/50 transition-transform hover:scale-110"
            >
              <Play className="h-5 w-5 fill-current ml-0.5" />
            </div>
          </div>
        </div>

        {/* Content */}
        <div className="p-5 space-y-4 flex flex-col flex-1">
          <div className="flex items-start justify-between">
            <div>
              <h3 className="font-semibold text-lg text-white group-hover:text-purple-400 transition-colors">
                {camera.name}
              </h3>
              <p className="text-xs text-gray-500">ID: CAM-{String(camera.id).padStart(3, '0')}</p>
            </div>
          </div>

          {/* Motion Summary */}
          <div className="space-y-2 rounded-lg bg-white/5 p-3 border border-white/5 flex-1">
            <div className="flex items-center gap-2 text-xs font-medium text-gray-400 uppercase tracking-wider mb-2">
              <Activity className="w-3 h-3 text-purple-400" />
              Ãšltimo movimiento
            </div>
            
            <div className="grid grid-cols-2 gap-2 text-sm">
              <div className="flex items-center gap-2 text-gray-300">
                <Clock className="w-3.5 h-3.5 text-gray-500" />
                {formatDate(camera.lastDetection?.timestamp)}
              </div>
              <div className="flex items-center gap-2 text-gray-300">
                <span className="w-1.5 h-1.5 rounded-full bg-orange-500" />
                {camera.lastDetection?.duration}s
              </div>
            </div>
          </div>

          <div className="grid grid-cols-2 gap-2">
            <Button 
              onClick={handleWatchLive}
              className="w-full bg-gray-800 hover:bg-gray-700 text-white border border-gray-700"
            >
              Ver en vivo
            </Button>
            <Button 
              onClick={handleUploadClick}
              variant="outline"
              className="w-full border-gray-700 hover:bg-gray-800 hover:text-white"
            >
              <Upload className="w-4 h-4 mr-2" />
              {isStoredVideo ? "Cambiar" : "Subir"}
            </Button>
          </div>
        </div>
      </div>
    </>
  );
};

export default CameraCard;
